#### 反转列表实现

    listNode* ReverseNList（listNonde* pHead）{
        listNonde* root=pHead;
        listNonde* pre=NULL;
        listNonde* next=NULL;
        if (pHead==NULL) return NULL;
        while(root->next){
            next=root->next;
            root->next=pre;
            pre=root;
            root=next;
        }
        root->next=pre;
        return root;
    }
